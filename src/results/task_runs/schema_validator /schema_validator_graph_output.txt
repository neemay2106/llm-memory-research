# Implementation: Type Coercion Handlers

I will now implement the complete CSV validator with type coercion and error accumulation:

```typescript
/**
 * CSV Schema Validator with Type Coercion and Error Accumulation
 * 
 * Design principles:
 * - Schema-driven validation
 * - All errors accumulated per row/field
 * - Type coercion for common CSV types
 * - Custom coercer support
 * - Efficient parsing for large files
 */

// ============================================================================
// TYPE DEFINITIONS
// ============================================================================

export type CoercibleType = 'string' | 'number' | 'boolean' | 'date';

export interface CoercionResult<T> {
  success: boolean;
  value?: T;
  error?: string;
}

export interface FieldSchema {
  name: string;
  type: CoercibleType;
  required?: boolean;
  custom?: (value: string) => CoercionResult<any>;
}

export interface RowValidationError {
  rowIndex: number;
  fieldName: string;
  originalValue: string;
  error: string;
}

export interface ValidationResult {
  valid: boolean;
  rowCount: number;
  errorCount: number;
  errors: RowValidationError[];
  data?: Map<number, Map<string, any>>; // rowIndex -> fieldName -> coercedValue
}

// ============================================================================
// COERCION HANDLERS - Efficient implementations
// ============================================================================

class CoercionHandlers {
  /**
   * String coercion: trimming and validation
   * Efficient: O(n) where n is string length, minimal allocations
   */
  static coerceString(value: string): CoercionResult<string> {
    const trimmed = value.trim();
    if (trimmed.length === 0) {
      return { success: false, error: 'String cannot be empty' };
    }
    return { success: true, value: trimmed };
  }

  /**
   * Number coercion: using Number() + validation
   * Efficient: Direct native parsing, fails on NaN
   */
  static coerceNumber(value: string): CoercionResult<number> {
    const trimmed = value.trim();
    
    // Handle empty strings
    if (trimmed === '') {
      return { success: false, error: 'Number cannot be empty' };
    }

    // Fast path: single numeric operations are optimized by JS engines
    const num = Number(trimmed);
    
    if (isNaN(num)) {
      return { success: false, error: `"${trimmed}" is not a valid number` };
    }

    return { success: true, value: num };
  }

  /**
   * Boolean coercion: strict common representations
   * Efficient: Direct string comparison (no regex), O(1)
   */
  static coerceBoolean(value: string): CoercionResult<boolean> {
    const trimmed = value.trim().toLowerCase();

    // Whitelist approach: faster than regex for small string set
    if (trimmed === 'true' || trimmed === '1' || trimmed === 'yes') {
      return { success: true, value: true };
    }
    if (trimmed === 'false' || trimmed === '0' || trimmed === 'no') {
      return { success: true, value: false };
    }

    return {
      success: false,
      error: `"${trimmed}" is not a valid boolean (true/false, 1/0, yes/no)`,
    };
  }

  /**
   * Date coercion: ISO 8601 and common formats
   * Efficient: Try fast path first (ISO), fall back to Date constructor
   */
  static coerceDate(value: string): CoercionResult<Date> {
    const trimmed = value.trim();

    if (trimmed === '') {
      return { success: false, error: 'Date cannot be empty' };
    }

    // Fast path: ISO 8601 format (most common in CSV exports)
    // Regex is unavoidable for robust date parsing, but we minimize it
    const isoPattern = /^\d{4}-\d{2}-\d{2}(T|Z|\s|$)/;
    if (isoPattern.test(trimmed)) {
      const date = new Date(trimmed);
      if (!isNaN(date.getTime())) {
        return { success: true, value: date };
      }
    }

    // Fallback: Try Date constructor on other formats
    const date = new Date(trimmed);
    if (isNaN(date.getTime())) {
      return {
        success: false,
        error: `"${trimmed}" is not a valid date (use ISO 8601 format YYYY-MM-DD)`,
      };
    }

    return { success: true, value: date };
  }
}

// ============================================================================
// SCHEMA-DRIVEN VALIDATOR
// ============================================================================

export class CSVValidator {
  private schema: FieldSchema[];
  private coercionMap: Map<CoercibleType, (val: string) => CoercionResult<any>>;

  constructor(schema: FieldSchema[]) {
    this.schema = schema;
    this.coercionMap = new Map([
      ['string', CoercionHandlers.coerceString],
      ['number', CoercionHandlers.coerceNumber],
      ['boolean', CoercionHandlers.coerceBoolean],
      ['date', CoercionHandlers.coerceDate],
    ]);
  }

  /**
   * Register a custom coercer for a field
   * Allows extension without modifying core handlers
   */
  registerCustomCoercer(fieldName: string, coercer: (val: string) => CoercionResult<any>): void {
    const field = this.schema.find((f) => f.name === fieldName);
    if (field) {
      field.custom = coercer;
    }
  }

  /**
   * Validate and coerce a single field value
   * Returns { success, value, error } following CoercionResult pattern
   */
  private coerceField(
    fieldSchema: FieldSchema,
    rawValue: string
  ): CoercionResult<any> {
    // Empty value handling
    if (rawValue.trim() === '') {
      if (fieldSchema.required !== false) {
        return { success: false, error: `${fieldSchema.name} is required` };
      }
      return { success: true, value: null };
    }

    // Use custom coercer if provided
    if (fieldSchema.custom) {
      return fieldSchema.custom(rawValue);
    }

    // Use built-in coercer for type
    const coercer = this.coercionMap.get(fieldSchema.type);
    if (!coercer) {
      return { success: false, error: `Unknown type: ${fieldSchema.type}` };
    }

    return coercer(rawValue);
  }

  /**
   * Validate a single row against the schema
   * Accumulates all field-level errors for this row
   * Returns tuple: [coercedRow, rowErrors]
   */
  private validateRow(
    rowIndex: number,
    values: string[]
  ): [Map<string, any> | null, RowValidationError[]] {
    const rowErrors: RowValidationError[] = [];
    const coercedRow = new Map<string, any>();

    // Validate each field in schema
    for (let fieldIdx = 0; fieldIdx < this.schema.length; fieldIdx++) {
      const fieldSchema = this.schema[fieldIdx];
      const rawValue = values[fieldIdx] ?? '';

      const result = this.coerceField(fieldSchema, rawValue);

      if (!result.success) {
        rowErrors.push({
          rowIndex,
          fieldName: fieldSchema.name,
          originalValue: rawValue,
          error: result.error!,
        });
      } else {
        coercedRow.set(fieldSchema.name, result.value);
      }
    }

    return rowErrors.length === 0 ? [coercedRow, rowErrors] : [null, rowErrors];
  }

  /**
   * Validate and coerce entire CSV data
   * Implements error accumulation: processes all rows, collects all errors
   * Returns ValidationResult with comprehensive error report
   */
  validate(rows: string[][]): ValidationResult {
    const errors: RowValidationError[] = [];
    const data = new Map<number, Map<string, any>>();

    // Process each row, accumulating errors
    for (let rowIdx = 0; rowIdx < rows.length; rowIdx++) {
      const [coercedRow, rowErrors] = this.validateRow(rowIdx, rows[rowIdx]);

      // Accumulate errors regardless of success/failure
      errors.push(...rowErrors);

      // Store coerced data only for successful rows
      if (coercedRow !== null) {
        data.set(rowIdx, coercedRow);
      }
    }

    const isValid = errors.length === 0;

    return {
      valid: isValid,
      rowCount: rows.length,
      errorCount: errors.length,
      errors,
      data: isValid ? data : undefined,
    };
  }
}

// Example: Define schema for a user CSV
const userSchema: FieldSchema[] = [
  { name: 'id', type: 'number', required: true },
  { name: 'name', type: 'string', required: true },
  { name: 'active', type: 'boolean', required: false },
  { name: 'joinDate', type: 'date', required: false },

// Create validator
const validator = new CSVValidator(userSchema);

// Example CSV data (as parsed strings)
const csvData: string[][] = [

  ['4', '', 'true', '2023-04-10'], // Missing required name

// Validate
const result = validator.validate(csvData);

console.log('=== VALIDATION RESULT ===');
console.log(`Valid: ${result.valid}`);
console.log(`Rows processed: ${result.rowCount}`);
console.log(`Errors found: ${result.errorCount}`);

if (result.errors.length > 0) {
  console.log('\n=== ERRORS ===');
  result.errors.forEach((err) => {
    console.log(
      `Row ${err.rowIndex}, Field "${err.fieldName}": ${err.error} (value: "${err.originalValue}")`
    );
  });
}

if (result.data) {
  console.log('\n=== COERCED DATA ===');
  result.data.forEach((row, rowIdx) => {
    console.log(`Row ${rowIdx}:`, Object.fromEntries(row));
  });
}

// Example: Custom coercer for email validation
validator.registerCustomCoercer('email', (value: string) => {
  const trimmed = value.trim();
  const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  if (emailPattern.test(trimmed)) {
    return { success: true, value: trimmed };
  }
  return { success: false, error: `"${trimmed}" is not a valid email` };
});
```

# Design Decisions Made in Implementation

1. **Error Accumulation (node_002 constraint):**
   - `validateRow()` processes all fields before returning, collecting all errors in a single pass
   - `validate()` processes all rows regardless of errors, no early exit
   - `ValidationResult` returns comprehensive error list with row/field/value context

2. **Performance (node_004 constraint):**
   - Boolean coercion uses direct string comparison (O(1)) instead of regex
   - Number coercion uses native `Number()` constructor (heavily optimized)
   - Date coercion tries ISO fast path first, minimizes regex usage
   - Single pass validation: O(rows × fields) with no redundant parsing

3. **Extensibility (node_003 decision):**
   - `registerCustomCoercer()` allows adding custom handlers without modifying core
   - `CoercionResult` interface standardizes return format across all handlers
   - Schema-driven design separates concerns
