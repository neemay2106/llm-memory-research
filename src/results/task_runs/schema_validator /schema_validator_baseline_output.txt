```python
"""
csv_validator/coercers.py

Type coercion handlers for CSV validation.
Each handler converts a string value to the target type and returns
(success: bool, value: any, error_message: str | None)
"""

from typing import Callable, Any, Optional, Tuple, Union
from datetime import datetime
import re


# Type signature for a coercer function
CoercerFunction = Callable[[str], Tuple[bool, Any, Optional[str]]]


class CoercionError(ValueError):
    """Raised when coercion fails with a recoverable error."""
    pass


class TypeCoercer:
    """
    Manages type coercion handlers for CSV fields.
    Supports built-in types (string, number, boolean, date) and custom coercers.
    """

    def __init__(self):
        self._coercers: dict[str, CoercerFunction] = {}
        self._register_builtin_coercers()

    def _register_builtin_coercers(self) -> None:
        """Register all built-in type coercers."""
        self.register("string", self._coerce_string)
        self.register("number", self._coerce_number)
        self.register("boolean", self._coerce_boolean)
        self.register("date", self._coerce_date)

    def register(self, type_name: str, coercer: CoercerFunction) -> None:
        """
        Register a custom type coercer.
        
        Args:
            type_name: Name of the type (e.g., "email", "uuid")
            coercer: Function(str) -> (bool, Any, Optional[str])
                    Returns (success, value, error_message)
        """
        if not callable(coercer):
            raise TypeError(f"Coercer must be callable, got {type(coercer)}")
        self._coercers[type_name] = coercer

    def coerce(self, value: str, type_name: str) -> Tuple[bool, Any, Optional[str]]:
        """
        Coerce a string value to the specified type.
        
        Args:
            value: String value from CSV
            type_name: Target type name
            
        Returns:
            (success, coerced_value, error_message)
            - success: True if coercion succeeded
            - coerced_value: The coerced value, or original string if failed
            - error_message: Human-readable error if failed, else None
        """
        if type_name not in self._coercers:
            return False, value, f"Unknown type: {type_name}"
        
        try:
            return self._coercers[type_name](value)
        except Exception as e:
            return False, value, f"Coercion error: {str(e)}"

    # =========================================================================
    # Built-in Coercers
    # =========================================================================

    @staticmethod
    def _coerce_string(value: str) -> Tuple[bool, str, None]:
        """
        String coercion: strips whitespace, always succeeds.
        """
        return True, value.strip(), None

    @staticmethod
    def _coerce_number(value: str) -> Tuple[bool, Union[int, float], Optional[str]]:
        """
        Number coercion: converts to int or float.
        Tries int first; falls back to float if decimal point present.
        """
        value = value.strip()
        
        # Empty string is invalid
        if not value:
            return False, None, "Empty value cannot be coerced to number"
        
        # Try int first (more efficient)
        try:
            return True, int(value), None
        except ValueError:
            pass
        
        # Try float
        try:
            return True, float(value), None
        except ValueError:
            return False, value, f"'{value}' is not a valid number"

    @staticmethod
    def _coerce_boolean(value: str) -> Tuple[bool, bool, Optional[str]]:
        """
        Boolean coercion: recognizes common true/false representations.
        Case-insensitive.
        """
        value_lower = value.strip().lower()
        
        if value_lower in ("true", "yes", "1", "on"):
            return True, True, None
        elif value_lower in ("false", "no", "0", "off"):
            return True, False, None
        else:
            return False, value, f"'{value}' is not a valid boolean"

    @staticmethod
    def _coerce_date(value: str) -> Tuple[bool, Optional[datetime], Optional[str]]:
        """
        Date coercion: attempts common ISO 8601 and US formats.
        Returns datetime object (or date-only if no time component).
        """
        value = value.strip()
        
        # List of common date formats to try
        # Ordered by likelihood (ISO 8601 first)
        formats = [
            "%Y-%m-%d",              # 2024-01-15
            "%Y-%m-%d %H:%M:%S",     # 2024-01-15 14:30:45
            "%Y-%m-%dT%H:%M:%S",     # 2024-01-15T14:30:45 (ISO 8601)
            "%m/%d/%Y",              # 01/15/2024 (US)
            "%m-%d-%Y",              # 01-15-2024
            "%d/%m/%Y",              # 15/01/2024 (EU)
            "%B %d, %Y",             # January 15, 2024
            "%b %d, %Y",             # Jan 15, 2024
        ]
        
        for fmt in formats:
            try:
                parsed = datetime.strptime(value, fmt)
                return True, parsed, None
            except ValueError:
                continue
        
        return False, value, f"'{value}' does not match any recognized date format"


# Singleton instance for global use
_default_coercer = TypeCoercer()


def get_default_coercer() -> TypeCoercer:
    """Get the default global TypeCoercer instance."""
    return _default_coercer
```

```python
"""
csv_validator/schema.py

Schema definition for CSV validation.
Defines columns, their types, constraints, and optional metadata.
"""

from typing import Optional, List, Dict, Any
from dataclasses import dataclass, field


@dataclass
class ColumnSchema:
    """
    Defines validation rules for a single CSV column.
    """
    name: str
    type_name: str
    required: bool = True
    default: Optional[Any] = None
    description: Optional[str] = None
    custom_metadata: Dict[str, Any] = field(default_factory=dict)
    
    def __post_init__(self):
        if not self.name:
            raise ValueError("Column name cannot be empty")
        if not self.type_name:
            raise ValueError("Type name cannot be empty")


@dataclass
class CSVSchema:
    """
    Complete schema for CSV validation.
    Maps column names to their validation rules.
    """
    columns: List[ColumnSchema]
    strict_mode: bool = False  # If True, reject unknown columns
    
    def __post_init__(self):
        if not self.columns:
            raise ValueError("Schema must contain at least one column")
        
        # Check for duplicate column names
        names = [col.name for col in self.columns]
        if len(names) != len(set(names)):
            raise ValueError("Duplicate column names in schema")
    
    def get_column(self, name: str) -> Optional[ColumnSchema]:
        """Get column schema by name."""
        for col in self.columns:
            if col.name == name:
                return col
        return None
    
    def column_names(self) -> List[str]:
        """Return list of column names in schema."""
        return [col.name for col in self.columns]
```

```python
"""
csv_validator/errors.py

Error accumulation and reporting for CSV validation.
"""

from typing import List, Dict, Any, Optional
from dataclasses import dataclass, field
from enum import Enum


class ErrorSeverity(Enum):
    """Severity level for validation errors."""
    WARNING = "warning"
    ERROR = "error"


@dataclass
class FieldError:
    """
    A single field-level validation error.
    """
    row_number: int
    column_name: str
    value: Any
    error_message: str
    severity: ErrorSeverity = ErrorSeverity.ERROR
    
    def __str__(self) -> str:
        return (
            f"Row {self.row_number}, Column '{self.column_name}': "
            f"{self.error_message} (value: {repr(self.value)})"
        )


@dataclass
class RowError:
    """
    All errors for a single row.
    """
    row_number: int
    field_errors: List[FieldError] = field(default_factory=list)
    
    def add_error(self, field_error: FieldError) -> None:
        """Add a field error to this row."""
        self.field_errors.append(field_error)
    
    def has_errors(self) -> bool:
        """Return True if this row has any errors."""
        return len(self.field_errors) > 0
    
    def error_count(self) -> int:
        """Return number of errors in this row."""
        return len(self.field_errors)


@dataclass
class ValidationResult:
    """
    Complete validation result with accumulated errors.
    """
    valid: bool
    row_errors: List[RowError] = field(default_factory=list)
    coerced_rows: List[Dict[str, Any]] = field(default_factory=list)
    
    def add_row_error(self, row_error: RowError) -> None:
        """Add a row error."""
        self.row_errors.append(row_error)
    
    def add_coerced_row(self, row_data: Dict[str, Any]) -> None:
        """Add a successfully coerced row."""
        self.coerced_rows.append(row_data)
    
    def total_errors(self) -> int:
        """Return total number of errors across all rows."""
        return sum(row.error_count() for row in self.row_errors)
    
    def error_summary(self) -> str:
        """Return human-readable summary of all errors."""
        if not self.row_errors:
            return "No errors"
        
        lines = [f"Validation failed with {self.total_errors()} error(s):"]
        for row_error in self.row_errors:
            for field_error in row_error.field_errors:
                lines.append(f"  {str(field_error)}")
        
        return "\n".join(lines)
    
    def is_valid(self) -> bool:
        """Return True if validation succeeded (no errors)."""
        return not self.row_errors
```
