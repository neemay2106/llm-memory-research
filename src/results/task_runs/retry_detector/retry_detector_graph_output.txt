```python
"""
Retry decorator with exponential backoff.

Implements a decorator that automatically retries a function with exponential
backoff, handling exceptions appropriately based on configuration.
"""

import functools
import time
from typing import Callable, Type, Tuple, TypeVar, Any

# Type variable for decorated function
F = TypeVar('F', bound=Callable[..., Any])


def retry(
    max_retries: int = 3,
    base_delay: float = 1.0,
    max_delay: float = 60.0,
    base: float = 2.0,
    exception_types: Tuple[Type[Exception], ...] = (Exception,)
) -> Callable[[F], F]:
    """
    Decorator that retries a function with exponential backoff.
    
    Args:
        max_retries: Maximum number of retry attempts (default: 3)
        base_delay: Initial delay in seconds (default: 1.0)
        max_delay: Maximum delay between retries in seconds (default: 60.0)
        base: Base for exponential calculation (default: 2.0)
        exception_types: Tuple of exception types to catch and retry on
                        (default: (Exception,))
    
    Returns:
        Decorated function that implements retry logic with exponential backoff
    
    Raises:
        The last exception encountered if max_retries is exhausted
    
    Example:
        @retry(max_retries=3, base_delay=1.0, max_delay=30.0)
        def flaky_api_call():
            # Automatically retried up to 3 times with exponential backoff
            pass
    """
    
    def decorator(func: F) -> F:
        @functools.wraps(func)
        def wrapper(*args, **kwargs) -> Any:
            last_exception = None
            
            for attempt in range(max_retries + 1):
                try:
                    # Attempt the function call
                    return func(*args, **kwargs)
                
                except exception_types as e:
                    last_exception = e
                    
                    # If we've exhausted retries, re-raise the exception
                    if attempt == max_retries:
                        raise
                    
                    # Calculate exponential backoff delay
                    delay = min(base_delay * (base ** attempt), max_delay)
                    
                    print(
                        f"Attempt {attempt + 1} failed with {type(e).__name__}: {e}. "
                        f"Retrying in {delay:.2f} seconds... "
                        f"(attempt {attempt + 1}/{max_retries})"
                    )
                    
                    time.sleep(delay)
        
        return wrapper  # type: ignore
    
    return decorator
```

"""
Test suite for retry decorator with exponential backoff.

import pytest
import time
from unittest.mock import patch, MagicMock


class TestRetryDecorator:

    def test_successful_call_no_retry(self):
        """A successful call should not trigger retries."""
        call_count = 0
        
        @retry(max_retries=3, base_delay=0.01)
        def successful_func():
            nonlocal call_count
            call_count += 1
            return "success"
        
        result = successful_func()
        assert result == "success"
        assert call_count == 1
    
    def test_retries_after_transient_failure(self):
        """Transient failures should trigger retries until success."""
        call_count = 0
        
        @retry(max_retries=3, base_delay=0.01)
        def flaky_func():
            nonlocal call_count
            call_count += 1
            if call_count < 3:
                raise RuntimeError("Transient error")
            return "success"
        
        result = flaky_func()
        assert result == "success"
        assert call_count == 3
    
    def test_exhausts_retries_and_reraises(self):
        """After max_retries, the exception should be re-raised."""
        call_count = 0
        
        @retry(max_retries=2, base_delay=0.01)
        def always_fails():
            nonlocal call_count
            call_count += 1
            raise ValueError("Permanent error")
        
        with pytest.raises(ValueError, match="Permanent error"):
            always_fails()
        
        # Should have tried: initial + 2 retries = 3 attempts
        assert call_count == 3
    
    def test_exponential_backoff_calculation(self):
        """Verify exponential backoff delays are calculated correctly."""
        call_count = 0
        call_times = []
        
        @retry(max_retries=2, base_delay=1.0, max_delay=10.0, base=2.0)
        def failing_func():
            nonlocal call_count
            call_times.append(time.time())
            call_count += 1
            if call_count <= 2:
                raise RuntimeError("Fail")
            return "success"
        
        result = failing_func()
        assert result == "success"
        assert call_count == 3
        
        # Check delays: attempt 0->1 should be ~1s, attempt 1->2 should be ~2s
        if len(call_times) >= 2:
            delay_1 = call_times[1] - call_times[0]
            assert 0.9 < delay_1 < 1.2, f"First delay {delay_1} not ~1s"
        
        if len(call_times) >= 3:
            delay_2 = call_times[2] - call_times[1]
            assert 1.9 < delay_2 < 2.2, f"Second delay {delay_2} not ~2s"
    
    def test_max_delay_ceiling(self):
        """Max_delay should cap the exponential growth."""
        call_count = 0
        
        @retry(max_retries=5, base_delay=1.0, max_delay=3.0, base=10.0)
        def failing_func():
            nonlocal call_count
            call_count += 1
            if call_count <= 5:
                raise RuntimeError("Fail")
            return "success"
        
        result = failing_func()
        assert result == "success"
        
        # At attempt 2: delay = min(1 * 10^2, 3) = min(100, 3) = 3.0
        # Should never exceed max_delay of 3.0
    
    def test_selective_exception_handling(self):
        """Only specified exception types should trigger retries."""
        call_count = 0
        
        @retry(max_retries=3, base_delay=0.01, exception_types=(ValueError,))
        def selective_retry():
            nonlocal call_count
            call_count += 1
            if call_count == 1:
                raise ValueError("Retryable")
            if call_count == 2:
                raise RuntimeError("Not retryable")
        
        with pytest.raises(RuntimeError, match="Not retryable"):
            selective_retry()
        
        # Should only call twice: once for ValueError, once for RuntimeError (not retried)
        assert call_count == 2
    
    def test_custom_parameters(self):
        """Decorator should accept custom parameters."""
        @retry(
            max_retries=5,
            base_delay=0.5,
            max_delay=20.0,
            base=3.0,
            exception_types=(IOError, TimeoutError)
        )
        def custom_retry_func():
            return "configured"
        
        result = custom_retry_func()
        assert result == "configured"
