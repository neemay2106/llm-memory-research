class Node:
    """
    Node class for Binary Search Tree.
    
    Stores a value and maintains left and right child pointers.
    This is the foundation for all BST operations.
    """
    def __init__(self, value):
        self.value = value
        self.left = None
        self.right = None


class BinarySearchTree:
    """
    Binary Search Tree implementation with insert, search, and delete operations.
    
    All operations maintain the BST property:
    - Left subtree: values < node value
    - Right subtree: values >= node value
    """
    
    def __init__(self):
        self.root = None
    
    # ==================== INSERT ====================
    def insert(self, value):
        """
        Insert a value into the BST.
        
        Uses recursive comparison and placement logic to maintain BST property.
        - Values less than node go left
        - Values greater than or equal go right
        
        Time Complexity: O(log n) average, O(n) worst case
        """
        if self.root is None:
            self.root = Node(value)
        else:
            self._insert_recursive(self.root, value)
    
    def _insert_recursive(self, node, value):
        """Helper method for recursive insertion."""
        if value < node.value:
            if node.left is None:
                node.left = Node(value)
            else:
                self._insert_recursive(node.left, value)
        else:
            if node.right is None:
                node.right = Node(value)
            else:
                self._insert_recursive(node.right, value)
    
    # ==================== SEARCH ====================
    def search(self, value):
        """
        Search for a value in the BST.
        
        Uses recursive traversal and comparison, leveraging the BST property
        for efficient O(log n) average case lookup.
        
        Returns: True if value found, False otherwise
        Time Complexity: O(log n) average, O(n) worst case
        """
        return self._search_recursive(self.root, value)
    
    def _search_recursive(self, node, value):
        """Helper method for recursive search."""
        if node is None:
            return False
        
        if value == node.value:
            return True
        elif value < node.value:
            return self._search_recursive(node.left, value)
        else:
            return self._search_recursive(node.right, value)
    
    # ==================== DELETE ====================
    def delete(self, value):
        """
        Delete a value from the BST.
        
        Handles three distinct cases:
        1. Leaf node: simply remove
        2. Node with one child: replace node with that child
        3. Node with two children: find inorder successor (leftmost node 
           in right subtree), replace value, then recursively delete successor
        
        Time Complexity: O(log n) average, O(n) worst case
        """
        self.root = self._delete_recursive(self.root, value)
    
    def _delete_recursive(self, node, value):
        """
        Helper method for recursive deletion.
        
        Returns the modified subtree after deletion.
        """
        if node is None:
            return None
        
        if value < node.value:
            # Recursively delete from left subtree
            node.left = self._delete_recursive(node.left, value)
        elif value > node.value:
            # Recursively delete from right subtree
            node.right = self._delete_recursive(node.right, value)
        else:
            # Found the node to delete
            
            # CASE 1: Leaf node (no children)
            if node.left is None and node.right is None:
                return None
            
            # CASE 2: Node with only right child
            if node.left is None:
                return node.right
            
            # CASE 2: Node with only left child
            if node.right is None:
                return node.left
            
            # CASE 3: Node with two children
            # Find inorder successor: leftmost node in right subtree
            successor = self._find_min(node.right)
            # Replace node's value with successor's value
            node.value = successor.value
            # Delete the successor (which has at most one child)
            node.right = self._delete_recursive(node.right, successor.value)
        
        return node
    
    def _find_min(self, node):
        """
        Find the node with minimum value in a subtree.
        
        Used to locate the inorder successor during deletion of nodes
        with two children.
        """
        current = node
        while current.left is not None:
            current = current.left
        return current
    
    # ==================== UTILITY METHODS ====================
    def inorder_traversal(self):
        """
        Perform inorder traversal (Left, Root, Right).
        
        Returns a list of values in sorted order.
        Useful for testing and visualization.
        """
        result = []
        self._inorder_recursive(self.root, result)
        return result
    
    def _inorder_recursive(self, node, result):
        """Helper method for inorder traversal."""
        if node is not None:
            self._inorder_recursive(node.left, result)
            result.append(node.value)
            self._inorder_recursive(node.right, result)
    
    def is_empty(self):
        """Check if the tree is empty."""
        return self.root is None
