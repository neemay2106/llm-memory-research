class Node:
    """
    Node class for Binary Search Tree.
    Decision: node_001 - Define the Node class structure with value, left, and right pointers.
    """
    def __init__(self, value):
        self.value = value
        self.left = None
        self.right = None


class BinarySearchTree:
    """
    Binary Search Tree implementation with insert, search, and delete operations.
    """
    
    def __init__(self):
        self.root = None
    
    # ========== INSERT ==========
    def insert(self, value):
        """
        Insert a value into the BST.
        Decision: node_002 - Implement insert with recursive comparison and placement logic.
        
        Maintains BST property: values < node go left, values >= node go right.
        Time Complexity: O(log n) average, O(n) worst case.
        """
        if self.root is None:
            self.root = Node(value)
        else:
            self._insert_recursive(self.root, value)
    
    def _insert_recursive(self, node, value):
        """Recursively insert value maintaining BST property."""
        if value < node.value:
            if node.left is None:
                node.left = Node(value)
            else:
                self._insert_recursive(node.left, value)
        else:  # value >= node.value (greater or equal go right)
            if node.right is None:
                node.right = Node(value)
            else:
                self._insert_recursive(node.right, value)
    
    # ========== SEARCH ==========
    def search(self, value):
        """
        Search for a value in the BST.
        Decision: node_003 - Implement search with recursive traversal and comparison.
        
        Leverages BST property for efficient lookup.
        Time Complexity: O(log n) average, O(n) worst case.
        
        Returns: True if value is found, False otherwise.
        """
        return self._search_recursive(self.root, value)
    
    def _search_recursive(self, node, value):
        """Recursively search for value in the tree."""
        if node is None:
            return False
        
        if value == node.value:
            return True
        elif value < node.value:
            return self._search_recursive(node.left, value)
        else:
            return self._search_recursive(node.right, value)
    
    # ========== DELETE ==========
    def delete(self, value):
        """
        Delete a value from the BST.
        Decision: node_004 - Implement delete with three cases: leaf nodes, one child, two children.
        
        Case 1: Leaf node - simply remove.
        Case 2: One child - replace node with that child.
        Case 3: Two children - find inorder successor (leftmost in right subtree),
                replace value, then recursively delete successor.
        
        Time Complexity: O(log n) average, O(n) worst case.
        """
        self.root = self._delete_recursive(self.root, value)
    
    def _delete_recursive(self, node, value):
        """Recursively delete value from the tree."""
        if node is None:
            return None
        
        if value < node.value:
            # Recursively delete from left subtree
            node.left = self._delete_recursive(node.left, value)
        elif value > node.value:
            # Recursively delete from right subtree
            node.right = self._delete_recursive(node.right, value)
        else:
            # Node to be deleted found
            
            # CASE 1: Leaf node (no children)
            if node.left is None and node.right is None:
                return None
            
            # CASE 2: Node with only one child
            if node.left is None:
                # Right child exists
                return node.right
            if node.right is None:
                # Left child exists
                return node.left
            
            # CASE 3: Node with two children
            # Find inorder successor (leftmost node in right subtree)
            successor = self._find_min(node.right)
            # Replace node's value with successor's value
            node.value = successor.value
            # Recursively delete the successor from right subtree
            node.right = self._delete_recursive(node.right, successor.value)
        
        return node
    
    def _find_min(self, node):
        """
        Find the node with minimum value (leftmost node).
        Used to locate inorder successor in delete operation.
        """
        current = node
        while current.left is not None:
            current = current.left
        return current
    
    # ========== UTILITY METHODS ==========
    def inorder_traversal(self):
        """
        Return list of values in inorder traversal (sorted order for BST).
        Useful for verification and debugging.
        """
        result = []
        self._inorder_recursive(self.root, result)
        return result
    
    def _inorder_recursive(self, node, result):
        """Recursively traverse in inorder: left -> node -> right."""
        if node is not None:
            self._inorder_recursive(node.left, result)
            result.append(node.value)
            self._inorder_recursive(node.right, result)
