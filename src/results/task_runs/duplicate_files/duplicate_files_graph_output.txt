```python
import hashlib
import os
from collections import defaultdict
from pathlib import Path


def find_duplicate_files(directory_path):
    """
    Find all duplicate files in a directory based on content hash.
    
    Uses hashlib (SHA256) with chunked file reading (8KB) to efficiently
    handle large files without consuming excessive memory.
    
    Args:
        directory_path (str or Path): Root directory to scan for duplicates
        
    Returns:
        dict: Mapping of file hash -> list of file paths with that hash.
              Only hashes with 2+ files are included.
              
    Raises:
        ValueError: If directory_path does not exist or is not a directory
    """
    directory_path = Path(directory_path)
    
    # Validate input
    if not directory_path.exists():
        raise ValueError(f"Directory does not exist: {directory_path}")
    if not directory_path.is_dir():
        raise ValueError(f"Path is not a directory: {directory_path}")
    
    # Dictionary to accumulate hashes -> file paths
    hash_to_files = defaultdict(list)
    
    # os.walk() traversal (node_002)
    for dirpath, dirnames, filenames in os.walk(directory_path):
        for filename in filenames:
            filepath = Path(dirpath) / filename
            
            try:
                # Compute hash using hashlib (node_001) with chunked reading (node_004)
                file_hash = _compute_file_hash(filepath)
                hash_to_files[file_hash].append(str(filepath))
                
            except (OSError, IOError) as e:
                # Log unreadable files but continue processing others
                print(f"Warning: Could not read {filepath}: {e}")
                continue
    
    # Filter to only return duplicates (2+ files per hash)
    duplicates = {
        hash_value: files 
        for hash_value, files in hash_to_files.items() 
        if len(files) > 1
    }
    
    return duplicates


def _compute_file_hash(filepath, chunk_size=8192):
    """
    Compute SHA256 hash of file contents using chunked reading.
    
    Args:
        filepath (Path or str): Path to file
        chunk_size (int): Size of chunks to read (default 8KB per node_004)
        
    Returns:
        str: Hexadecimal hash string
    """
    sha256_hash = hashlib.sha256()
    
    # Chunked file reading (node_004: 8KB chunks for memory efficiency)
    with open(filepath, 'rb') as f:
        while True:
            chunk = f.read(chunk_size)
            if not chunk:
                break
            sha256_hash.update(chunk)
    
    return sha256_hash.hexdigest()


# Example usage

            for filepath in files:

    else:
        print("No duplicate files found.")
```
