```python
from collections import deque
from time import time
from threading import Lock, Thread
from typing import Optional


class RateLimiter:
    """
    Rate limiter implementing sliding window algorithm.
    
    Maintains request timestamps in a deque per client and removes
    expired timestamps both during request checks and periodically.
    """
    
    def __init__(self, max_requests: int, window_seconds: int = 60, 
                 periodic_cleanup_interval: Optional[float] = None):
        """
        Initialize the rate limiter.
        
        Args:
            max_requests: Maximum number of requests allowed per window
            window_seconds: Time window in seconds (default 60 for per-minute)
            periodic_cleanup_interval: Seconds between cleanup jobs (optional).
                                      If None, cleanup only happens during requests.
        """
        self.max_requests = max_requests
        self.window_seconds = window_seconds
        self.client_timestamps = {}  # Dict[client_id -> deque of timestamps]
        self.lock = Lock()  # Thread-safe access to shared state
        self.periodic_cleanup_interval = periodic_cleanup_interval
        self._cleanup_thread = None
        self._running = False
        
        # Start periodic cleanup if interval specified
        if periodic_cleanup_interval is not None:
            self._start_periodic_cleanup()
    
    def allow_request(self, client_id: str) -> bool:
        """
        Check if a client is allowed to make a request.
        
        Implements automatic cleanup of expired timestamps during the check.
        
        Args:
            client_id: Unique identifier for the client
            
        Returns:
            True if request is allowed, False if rate limit exceeded
        """
        with self.lock:
            current_time = time()
            
            # Initialize client if first request
            if client_id not in self.client_timestamps:
                self.client_timestamps[client_id] = deque()
            
            client_queue = self.client_timestamps[client_id]
            
            # --- AUTOMATIC CLEANUP: Remove expired timestamps ---
            # Remove all timestamps older than the sliding window
            while client_queue and client_queue[0] < current_time - self.window_seconds:
                client_queue.popleft()
            
            # Check if request is allowed
            if len(client_queue) < self.max_requests:
                # Add current timestamp and allow request
                client_queue.append(current_time)
                return True
            else:
                # Rate limit exceeded
                return False
    
    def _cleanup_idle_clients(self) -> None:
        """
        Remove clients with no recent activity.
        
        Only called by periodic cleanup thread. Removes clients whose
        most recent request is older than the window.
        """
        with self.lock:
            current_time = time()
            idle_clients = []
            
            # Identify clients with no requests in current window
            for client_id, client_queue in self.client_timestamps.items():
                if not client_queue or client_queue[-1] < current_time - self.window_seconds:
                    idle_clients.append(client_id)
            
            # Remove idle clients
            for client_id in idle_clients:
                del self.client_timestamps[client_id]
    
    def _start_periodic_cleanup(self) -> None:
        """Start the periodic cleanup background thread."""
        self._running = True
        self._cleanup_thread = Thread(target=self._periodic_cleanup_worker, daemon=True)
        self._cleanup_thread.start()
    
    def _periodic_cleanup_worker(self) -> None:
        """Background worker that periodically cleans up idle clients."""
        while self._running:
            try:
                # Sleep for the specified interval
                time.sleep(self.periodic_cleanup_interval)
                if self._running:  # Check again after sleep
                    self._cleanup_idle_clients()
            except Exception as e:
                # Log but don't crash the cleanup thread
                print(f"Error in periodic cleanup: {e}")
    
    def stop(self) -> None:
        """Stop the periodic cleanup thread (if running)."""
        self._running = False
        if self._cleanup_thread is not None:
            self._cleanup_thread.join(timeout=2.0)
    
    def get_client_request_count(self, client_id: str) -> int:
        """
        Get current request count for a client within the sliding window.
        
        Useful for monitoring and testing.
        """
        with self.lock:
            if client_id not in self.client_timestamps:
                return 0
            
            current_time = time()
            client_queue = self.client_timestamps[client_id]
            
            # Remove expired timestamps first (consistency)
            while client_queue and client_queue[0] < current_time - self.window_seconds:
                client_queue.popleft()
            
            return len(client_queue)
    
    def reset_client(self, client_id: str) -> None:
        """Reset all request history for a specific client."""
        with self.lock:
            if client_id in self.client_timestamps:
                self.client_timestamps[client_id].clear()
    
    def reset_all(self) -> None:
        """Reset all clients and their request history."""
        with self.lock:
            self.client_timestamps.clear()
    
    def __del__(self):
        """Cleanup on object destruction."""
        self.stop()
```

## 4. Implementation Details Aligned with Decisions

| Decision | Implementation |
|----------|-----------------|
| **Sliding Window Algorithm** | `allow_request()` maintains exact 60-second window; expired timestamps removed via `popleft()` |
| **Dict + Deque per Client** | `self.client_timestamps: Dict[str -> deque]` with O(1) lookup and O(1) removal |
| **Automatic Cleanup** | `while client_queue and client_queue[0] < current_time - self.window_seconds: client_queue.popleft()` runs on every `allow_request()` call |
| **Periodic Cleanup** | Optional `_periodic_cleanup_worker()` thread removes idle clients at configurable interval |
| **Thread Safety** | `Lock` protects all shared state access |
