def flatten_json(nested_dict, parent_key='', separator='.', max_depth=None):
    """
    Flatten a nested JSON structure into a single-level dictionary.
    
    Args:
        nested_dict (dict): The nested dictionary to flatten
        parent_key (str): The parent key (used internally for recursion)
        separator (str): The separator to use between nested keys (default: '.')
        max_depth (int): Maximum recursion depth. If None, uses Python's limit.
                        Set to prevent RecursionError on deeply nested structures.
                        Default Python recursion limit is typically ~1000.
    
    Returns:
        dict: A flattened single-level dictionary with dot-separated keys
    
    Raises:
        ValueError: If max_depth is exceeded during recursion
        TypeError: If input is not a dictionary
    
    Examples:
        >>> flatten_json({'a': {'b': 1, 'c': 2}})
        {'a.b': 1, 'a.c': 2}
        
        >>> flatten_json({'x': 1, 'y': {'z': 3}})
        {'x': 1, 'y.z': 3}
    
    KNOWN LIMITATIONS:
        1. Recursion Depth (node_002 constraint): Deeply nested structures 
           (>1000 levels) may raise RecursionError. Use max_depth parameter 
           to guard against this.
        
        2. Dot Notation Ambiguity (node_004 constraint): If original keys 
           contain dots, the result is ambiguous. Example:
           {'a.b': {'c': 1}} and {'a': {'b': {'c': 1}}} both flatten to 
           {'a.b.c': 1}. This limits reversibility.
    """
    if not isinstance(nested_dict, dict):
        raise TypeError(f"Expected dict, got {type(nested_dict).__name__}")
    
    if max_depth is not None and max_depth <= 0:
        raise ValueError("Recursion depth exceeded: max_depth must be > 0")
    
    items = []
    
    for key, value in nested_dict.items():
        # Build the new key using the separator
        new_key = f"{parent_key}{separator}{key}" if parent_key else key
        
        # Recurse if value is a dict
        if isinstance(value, dict):
            next_depth = max_depth - 1 if max_depth is not None else None
            if max_depth is not None and next_depth <= 0:
                raise ValueError(
                    f"Recursion depth exceeded at key '{new_key}'. "
                    f"Original max_depth={max_depth} was too shallow."
                )
            
            # Recursive call (node_001 decision: recursion as core algorithm)
            nested_items = flatten_json(
                value,
                parent_key=new_key,
                separator=separator,
                max_depth=next_depth
            )
            items.extend(nested_items.items())
        else:
            # Leaf node: add to result
            items.append((new_key, value))
    
    return dict(items)
