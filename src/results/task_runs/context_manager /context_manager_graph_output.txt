```python
import time
from typing import Optional

class CodeBlockTimer:
    """Context manager for timing code blocks with per-label accumulation."""
    
    # Class-level storage: {label: total_elapsed_seconds}
    timings: dict[str, float] = {}
    
    def __init__(self, label: str) -> None:
        """
        Initialize the timer with a label.
        
        Args:
            label: Identifier for the code block being timed
        """
        self.label = label
        self.start_time: Optional[float] = None
    
    def __enter__(self):
        """Record the start time when entering the context."""
        self.start_time = time.perf_counter()
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        """
        Calculate elapsed time and accumulate it in the class-level timings dictionary.
        
        Args:
            exc_type: Exception type if one was raised, else None
            exc_val: Exception value if one was raised, else None
            exc_tb: Exception traceback if one was raised, else None
        
        Returns:
            False (or None) to propagate any exception that occurred
        """
        elapsed = time.perf_counter() - self.start_time
        
        # Accumulate elapsed time under the label
        if self.label in self.timings:
            self.timings[self.label] += elapsed
        else:
            self.timings[self.label] = elapsed
        
        # Do not suppress exceptions
        return False
    
    @classmethod
    def summary(cls) -> str:
        """
        Generate a per-label summary report of accumulated timing data.
        
        Returns:
            A formatted string report showing each label and total accumulated time
        """
        if not cls.timings:
            return "No timing data recorded."
        
        # Sort labels alphabetically for consistent output
        lines = ["Timing Summary:"]
        for label in sorted(cls.timings.keys()):
            total_time = cls.timings[label]
            lines.append(f"  {label}: {total_time:.6f}s")
        
        return "\n".join(lines)
    
    @classmethod
    def reset(cls) -> None:
        """Clear all accumulated timing data."""
        cls.timings.clear()


# Example usage
if __name__ == "__main__":
    import time
    
    # Time multiple blocks
    with CodeBlockTimer("task_a"):
        time.sleep(0.1)
    
    with CodeBlockTimer("task_b"):
        time.sleep(0.05)
    
    with CodeBlockTimer("task_a"):
        time.sleep(0.1)
    
    # Print summary
    print(CodeBlockTimer.summary())
```
