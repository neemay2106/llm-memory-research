# Implementation

```javascript
class EventEmitter {
  constructor() {
    // Map: eventName (string) -> array of listener functions
    this.listeners = new Map();
  }

  /**
   * Register an event listener
   * @param {string} eventName - The name of the event
   * @param {Function} listener - The callback function to execute when the event is emitted
   * @returns {EventEmitter} - Returns this for method chaining
   */
  on(eventName, listener) {
    // Validate inputs
    if (typeof eventName !== 'string') {
      throw new TypeError('Event name must be a string');
    }
    if (typeof listener !== 'function') {
      throw new TypeError('Listener must be a function');
    }

    // Initialize listener array for this event if it doesn't exist
    if (!this.listeners.has(eventName)) {
      this.listeners.set(eventName, []);
    }

    // Add the listener to the array
    this.listeners.get(eventName).push(listener);

    // Return this for method chaining
    return this;
  }

  /**
   * Emit an event, calling all registered listeners with provided arguments
   * @param {string} eventName - The name of the event to emit
   * @param {...any} args - Arguments to pass to each listener
   * @returns {boolean} - Returns true if the event had listeners, false otherwise
   */
  emit(eventName, ...args) {
    // Validate input
    if (typeof eventName !== 'string') {
      throw new TypeError('Event name must be a string');
    }

    // Check if this event has any listeners
    if (!this.listeners.has(eventName)) {
      return false;
    }

    const eventListeners = this.listeners.get(eventName);

    // Call each listener with the provided arguments
    for (const listener of eventListeners) {
      listener(...args);
    }

    // Return true if listeners were called
    return eventListeners.length > 0;
  }

  /**
   * Remove a specific listener from an event
   * @param {string} eventName - The name of the event
   * @param {Function} listener - The specific listener function to remove
   * @returns {EventEmitter} - Returns this for method chaining
   */
  off(eventName, listener) {
    // Validate inputs
    if (typeof eventName !== 'string') {
      throw new TypeError('Event name must be a string');
    }
    if (typeof listener !== 'function') {
      throw new TypeError('Listener must be a function');
    }

    // If the event doesn't exist, return early
    if (!this.listeners.has(eventName)) {
      return this;
    }

    const eventListeners = this.listeners.get(eventName);

    // Find and remove the listener (linear search O(n) as documented)
    const index = eventListeners.indexOf(listener);
    if (index !== -1) {
      eventListeners.splice(index, 1);
    }

    // If no listeners remain, clean up the event entry
    if (eventListeners.length === 0) {
      this.listeners.delete(eventName);
    }

    // Return this for method chaining
    return this;
  }
}

// Export for use in modules
module.exports = EventEmitter;
```

# Usage Examples

```javascript
// Create an instance
const emitter = new EventEmitter();

// Register listeners with method chaining
emitter
  .on('user:login', (username) => {
    console.log(`${username} logged in`);
  })
  .on('user:login', (username) => {
    console.log(`Welcome back, ${username}!`);
  })
  .on('user:logout', (username) => {
    console.log(`${username} logged out`);
  });

// Emit events
emitter.emit('user:login', 'alice');
// Output:
// alice logged in
// Welcome back, alice!

emitter.emit('user:logout', 'alice');
// Output:
// alice logged out

// Remove a listener
const loginHandler = (username) => console.log(`${username} signed in`);
emitter.on('user:login', loginHandler);
emitter.off('user:login', loginHandler);

// Check if event has listeners
console.log(emitter.emit('user:login', 'bob')); // true (other handlers exist)
console.log(emitter.emit('nonexistent:event')); // false (no listeners)
```
