# Implementation

```python
class Node:
    """Node in the doubly linked list."""
    def __init__(self, key, value):
        self.key = key
        self.value = value
        self.prev = None
        self.next = None


class LRUCache:
    """
    LRU Cache implementation using HashMap + Doubly Linked List.
    
    Time Complexity:
    - get(key): O(1)
    - put(key, value): O(1)
    
    Space Complexity: O(capacity)
    """
    
    def __init__(self, capacity: int):
        """
        Initialize LRU cache with a given capacity.
        
        Args:
            capacity: Fixed capacity limit for the cache
        
        Raises:
            ValueError: If capacity is less than 1
        """
        if capacity < 1:
            raise ValueError("Capacity must be at least 1")
        
        self.capacity = capacity
        self.cache = {}  # HashMap: key -> Node
        
        # Dummy nodes for doubly linked list
        # Front (after head) = most recently used
        # Back (before tail) = least recently used
        self.head = Node(0, 0)
        self.tail = Node(0, 0)
        self.head.next = self.tail
        self.tail.prev = self.head
    
    def _remove_node(self, node: Node) -> None:
        """Remove a node from the doubly linked list."""
        prev_node = node.prev
        next_node = node.next
        prev_node.next = next_node
        next_node.prev = prev_node
    
    def _add_to_front(self, node: Node) -> None:
        """Add a node to the front (right after head) of the doubly linked list."""
        node.next = self.head.next
        node.prev = self.head
        self.head.next.prev = node
        self.head.next = node
    
    def get(self, key: int) -> int:
        """
        Retrieve the value associated with the key.
        Marks the accessed item as most recently used.
        
        Args:
            key: The key to retrieve
        
        Returns:
            The value associated with key, or -1 if key does not exist
        """
        if key not in self.cache:
            return -1
        
        node = self.cache[key]
        # Move accessed node to front (most recently used)
        self._remove_node(node)
        self._add_to_front(node)
        
        return node.value
    
    def put(self, key: int, value: int) -> None:
        """
        Insert or update a key-value pair in the cache.
        
        Handles three cases:
        1. Key exists: Update value and move to front
        2. Key doesn't exist, under capacity: Add to front
        3. Key doesn't exist, at capacity: Evict LRU (back) node, then add to front
        
        Args:
            key: The key to insert or update
            value: The value to associate with the key
        
        Enforces the capacity constraint by evicting the least recently used item
        when the cache is at capacity.
        """
        # Case 1: Key already exists - update value and move to front
        if key in self.cache:
            node = self.cache[key]
            node.value = value
            self._remove_node(node)
            self._add_to_front(node)
            return
        
        # Case 2 & 3: Key doesn't exist
        # Check if we need to evict (Case 3)
        if len(self.cache) >= self.capacity:
            # Evict the least recently used item (node before tail)
            lru_node = self.tail.prev
            self._remove_node(lru_node)
            del self.cache[lru_node.key]
        
        # Add new node to front (most recently used)
        new_node = Node(key, value)
        self.cache[key] = new_node
        self._add_to_front(new_node)
```

# Implementation Summary

**What was implemented:**

1. **Node class**: Represents a doubly linked list node with key, value, and pointers
2. **LRUCache class**:
   - `__init__`: Initializes with capacity, HashMap (dict), and dummy head/tail nodes
   - `_remove_node`: Removes node from linked list in O(1)
   - `_add_to_front`: Adds node to front (most recently used position) in O(1)
   - `get(key)`: Returns value, marks as recently used, returns -1 if not found
   - `put(key, value)`: Handles all three cases defined in node_004
     - **Case 1** (key exists): Update value and move to front
     - **Case 2** (key doesn't exist, under capacity): Add to front
     - **Case 3** (key doesn't exist, at capacity): Evict LRU (back node), then add to front

**Constraint Compliance:**

✅ **node_002 (blocking)**: Capacity constraint is actively enforced in put() — when `len(self.cache) >= self.capacity`, the LRU node (at back of linked list) is evicted before insertion.

✅ **edge_001**: The architecture respects the constraint — HashMap enables O(1) lookups, doubly linked list enables O(1) eviction of the back node.

**Time & Space Complexity:**
- `get(key)`: O(1)
- `put(key, value)`: O(1)
- Space: O(capacity)
