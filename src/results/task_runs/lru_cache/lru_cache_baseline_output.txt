```python
class Node:
    """Doubly Linked List Node for LRU Cache"""
    def __init__(self, key=0, value=0):
        self.key = key
        self.value = value
        self.prev = None
        self.next = None


class LRUCache:
    """
    LRU Cache implementation using HashMap + Doubly Linked List.
    
    - HashMap: O(1) lookups by key
    - Doubly Linked List: maintains access order, O(1) eviction from back
    
    Invariants:
    - Most recently used item is at the front (after head)
    - Least recently used item is at the back (before tail)
    - Size never exceeds capacity
    """
    
    def __init__(self, capacity: int):
        """
        Initialize LRU Cache with fixed capacity.
        
        Args:
            capacity: Maximum number of items the cache can hold
        """
        if capacity <= 0:
            raise ValueError("Capacity must be positive")
        
        self.capacity = capacity
        self.cache = {}  # Maps key -> Node
        
        # Dummy head and tail nodes to simplify edge cases
        self.head = Node()  # Most recently used side
        self.tail = Node()  # Least recently used side
        
        # Link dummy nodes
        self.head.next = self.tail
        self.tail.prev = self.head
    
    def _add_to_front(self, node: Node) -> None:
        """Add node right after head (most recently used position)"""
        node.prev = self.head
        node.next = self.head.next
        self.head.next.prev = node
        self.head.next = node
    
    def _remove_node(self, node: Node) -> None:
        """Remove node from doubly linked list"""
        prev_node = node.prev
        next_node = node.next
        prev_node.next = next_node
        next_node.prev = prev_node
    
    def _move_to_front(self, node: Node) -> None:
        """Move existing node to front (most recently used)"""
        self._remove_node(node)
        self._add_to_front(node)
    
    def _evict_lru(self) -> None:
        """Remove least recently used item (from back, before tail)"""
        lru_node = self.tail.prev
        self._remove_node(lru_node)
        del self.cache[lru_node.key]
    
    def get(self, key: int) -> int:
        """
        Get the value of a key if it exists in cache, otherwise return -1.
        
        Operations:
        1. Check if key exists in HashMap
        2. If found: move node to front (mark as most recently used)
        3. Return value or -1
        
        Time Complexity: O(1)
        
        Args:
            key: The key to retrieve
            
        Returns:
            The value associated with key, or -1 if not found
        """
        if key not in self.cache:
            return -1
        
        node = self.cache[key]
        self._move_to_front(node)
        return node.value
    
    def put(self, key: int, value: int) -> None:
        """
        Put key-value pair into cache, handling three cases:
        
        Case 1: Key exists → update value and move to front
        Case 2: Key doesn't exist, under capacity → create node, add to front
        Case 3: Key doesn't exist, at capacity → evict LRU, create node, add to front
        
        Time Complexity: O(1)
        
        Args:
            key: The key to insert/update
            value: The value to associate with the key
        """
        # Case 1: Key already exists - update and move to front
        if key in self.cache:
            node = self.cache[key]
            node.value = value
            self._move_to_front(node)
            return
        
        # Case 2 & 3: Key doesn't exist
        # If at capacity, evict LRU item first (Case 3)
        if len(self.cache) >= self.capacity:
            self._evict_lru()
        
        # Create new node and add to front (most recently used position)
        new_node = Node(key, value)
        self.cache[key] = new_node
        self._add_to_front(new_node)
```
