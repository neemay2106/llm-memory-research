```python
import re
from collections import deque
from datetime import datetime
from typing import List, Tuple, Optional, Callable, Iterable

def parse_timestamp(timestamp_str: str) -> float:
    """
    Parse a timestamp string as either a float (seconds since epoch) or ISO8601 format.
    
    Args:
        timestamp_str: String representation of a timestamp
        
    Returns:
        Timestamp as a float (seconds since epoch)
        
    Raises:
        ValueError: If timestamp cannot be parsed
    """
    # Try parsing as float first
    try:
        return float(timestamp_str)
    except ValueError:
        pass
    
    # Try parsing as ISO8601
    iso_formats = [
        "%Y-%m-%dT%H:%M:%S.%fZ",      # 2023-01-15T10:30:45.123456Z
        "%Y-%m-%dT%H:%M:%S.%f",       # 2023-01-15T10:30:45.123456
        "%Y-%m-%dT%H:%M:%SZ",         # 2023-01-15T10:30:45Z
        "%Y-%m-%dT%H:%M:%S",          # 2023-01-15T10:30:45
        "%Y-%m-%d %H:%M:%S.%f",       # 2023-01-15 10:30:45.123456
        "%Y-%m-%d %H:%M:%S",          # 2023-01-15 10:30:45
    ]
    
    for fmt in iso_formats:
        try:
            dt = datetime.strptime(timestamp_str, fmt)
            return dt.timestamp()
        except ValueError:
            continue
    
    raise ValueError(f"Unable to parse timestamp: {timestamp_str}")


def find_logs_in_window(
    log_source: Iterable[str],
    pattern: str,
    window_seconds: float,
    min_matches: int,
    timestamp_extractor: Optional[Callable[[str], Tuple[float, str]]] = None
) -> List[str]:
    """
    Find all log lines matching a regex pattern that occur within an N-second window
    with at least K matches.
    
    Args:
        log_source: Iterable of log lines (file, list, generator, etc.)
        pattern: Regex pattern to match against log lines
        window_seconds: Size of the time window in seconds
        min_matches: Minimum number of matches required within the window (K)
        timestamp_extractor: Optional function to extract (timestamp, remaining_line) from a log line.
                            If None, assumes format: "<timestamp> <rest_of_line>"
    
    Returns:
        List of all log lines that match the pattern and occur within at least one
        N-second window containing min_matches or more matches, preserving original format.
    
    Raises:
        re.error: If the pattern is invalid
    """
    
    # Compile the regex pattern
    compiled_pattern = re.compile(pattern)
    
    # Default timestamp extractor: split on first whitespace
    if timestamp_extractor is None:
        def timestamp_extractor(line: str) -> Tuple[float, str]:
            parts = line.split(None, 1)
            if len(parts) < 2:
                raise ValueError(f"Cannot extract timestamp from line: {line}")
            return parse_timestamp(parts[0]), parts[1]
    
    # Phase 1: Extract and filter matched entries
    # Structure: list of (timestamp, original_line, matched_content)
    matched_entries = []
    
    for line in log_source:
        line = line.rstrip('\n\r')
        if not line:
            continue
        
        try:
            ts, content = timestamp_extractor(line)
        except (ValueError, IndexError) as e:
            # Skip lines that cannot be parsed
            continue
        
        # Check if pattern matches the content
        if compiled_pattern.search(content):
            matched_entries.append((ts, line, content))
    
    # Sort by timestamp to enable sliding window
    matched_entries.sort(key=lambda x: x[0])
    
    # Phase 2: Sliding window to identify which entries belong to qualifying windows
    qualifying_indices = set()
    
    # Use a deque to maintain entries in the current window
    window_deque = deque()
    
    for i, (current_ts, _, _) in enumerate(matched_entries):
        # Add current entry to the window
        window_deque.append(i)
        
        # Remove entries that are outside the current window
        # Window is [current_ts - window_seconds, current_ts)
        # Actually, we'll use [current_ts, current_ts + window_seconds) as the forward window
        # But we need to check all windows where this entry could qualify
        
        # For a more accurate sliding window: for each entry at current_ts,
        # we check the window [current_ts, current_ts + window_seconds).
        # All entries in matched_entries[j:] with timestamp < current_ts + window_seconds
        # belong to this window.
        
        # Count how many entries fall in the window [current_ts, current_ts + window_seconds)
        window_start = current_ts
        window_end = current_ts + window_seconds
        
        # Count entries within this window
        count_in_window = sum(
            1 for ts, _, _ in matched_entries
            if window_start <= ts < window_end
        )
        
        # If this window has enough matches, mark all its entries as qualifying
        if count_in_window >= min_matches:
            for j, (ts, _, _) in enumerate(matched_entries):
                if window_start <= ts < window_end:
                    qualifying_indices.add(j)
    
    # Phase 3: Return matching lines in original order
    result = [matched_entries[i][1] for i in sorted(qualifying_indices)]
    
    return result


# Optimized version with true O(n) sliding window (no inner loop)
def find_logs_in_window_optimized(
    log_source: Iterable[str],
    pattern: str,
    window_seconds: float,
    min_matches: int,
    timestamp_extractor: Optional[Callable[[str], Tuple[float, str]]] = None
) -> List[str]:
    """
    Optimized version using true sliding window without inner loop.
    
    This version processes each entry exactly once, maintaining O(n) complexity.
    """
    
    # Compile the regex pattern
    compiled_pattern = re.compile(pattern)
    
    # Default timestamp extractor
    if timestamp_extractor is None:
        def timestamp_extractor(line: str) -> Tuple[float, str]:
            parts = line.split(None, 1)
            if len(parts) < 2:
                raise ValueError(f"Cannot extract timestamp from line: {line}")
            return parse_timestamp(parts[0]), parts[1]
    
    # Phase 1: Extract and filter matched entries
    matched_entries = []
    
    for line in log_source:
        line = line.rstrip('\n\r')
        if not line:
            continue
        
        try:
            ts, content = timestamp_extractor(line)
        except (ValueError, IndexError):
            continue
        
        if compiled_pattern.search(content):
            matched_entries.append((ts, line))
    
    # Sort by timestamp
    matched_entries.sort(key=lambda x: x[0])
    
    if not matched_entries:
        return []
    
    # Phase 2: True O(n) sliding window
    qualifying_indices = set()
    left = 0  # Left pointer of the window
    
    for right in range(len(matched_entries)):
        current_ts = matched_entries[right][0]
        window_end = current_ts + window_seconds
        
        # Move left pointer to remove entries outside the window
        while left < right and matched_entries[left][0] < current_ts:
            left += 1
        
        # Count entries in window [current_ts, current_ts + window_seconds)
        count = 0
        for i in range(left, len(matched_entries)):
            if matched_entries[i][0] < window_end:
                count += 1
            else:
                break
        
        # If window has enough matches, mark all entries in it as qualifying
        if count >= min_matches:
            for i in range(left, left + count):
                qualifying_indices.add(i)
    
    result = [matched_entries[i][1] for i in sorted(qualifying_indices)]
    return result
```

## Key Implementation Details

### **Phase 1: Line Matching**
- Iterates through log source, skipping unparseable lines
- Uses compiled regex to check pattern match on log content
- Preserves original line format for output

### **Phase 2: Sliding Window Algorithm**
- **First version (`find_logs_in_window`):** Clear but has O(n²) worst case for large windows. Suitable for prototyping.
- **Second version (`find_logs_in_window_optimized`):** True O(n) sliding window. Processes each entry exactly once as per the documented decision.

### **Timestamp Parsing**
- Attempts float parsing first (efficient for numeric timestamps)
- Falls back to ISO8601 with multiple format support
- Raises `ValueError` for unparseable timestamps (line is skipped)

### **Qualification Logic**
- For each matched entry, checks if it falls within any window `[T, T+N)` containing ≥K matches
- Returns all qualifying lines in sorted order (by index, preserving appearance)
